-- Creación de la tabla Usuario
CREATE TABLE Usuario (
    usuario_id INT PRIMARY KEY,
    nombre VARCHAR(120) NOT NULL,
    email VARCHAR(120) NOT NULL,
    contrasena VARCHAR(120) NOT NULL
);

-- Creación de la tabla Empleado
CREATE TABLE Empleado (
    empleado_id INT PRIMARY KEY,
    nombre VARCHAR(120) NOT NULL,
    apellido VARCHAR(120) NOT NULL,
    sexo CHAR(1) NOT NULL,
    edad INT NOT NULL,
    puesto VARCHAR(120) NOT NULL,
    estatura FLOAT NOT NULL,
    horas_trabajo DATETIME NOT NULL,
    horas_descanso DATETIME NOT NULL
);

-- Creación de la tabla Actividad
CREATE TABLE Actividad (
    actividad_id INT PRIMARY KEY,
    usuario_id INT,
    tipo_actividad VARCHAR(120) NOT NULL,
    descripcion TEXT NOT NULL,
    fecha_actividad DATETIME NOT NULL,
    FOREIGN KEY (usuario_id) REFERENCES Usuario(usuario_id)
);

-- Creación de la tabla Autenticacion
CREATE TABLE Autenticacion (
    autenticacion_id INT PRIMARY KEY,
    usuario_id INT,
    session_id VARCHAR(255) NOT NULL,
    inicio_sesion DATETIME NOT NULL,
    fin_sesion DATETIME NOT NULL,
    FOREIGN KEY (usuario_id) REFERENCES Usuario(usuario_id)
);

-- Creación de la tabla Resultado
CREATE TABLE Resultado (
    resultado_id INT PRIMARY KEY,
    actividad_id INT,
    indice_ocra FLOAT NOT NULL,
    comentarios TEXT,
    FOREIGN KEY (actividad_id) REFERENCES Actividad(actividad_id)
);

-- Creación de la tabla Factor
CREATE TABLE Factor (
    factor_id INT PRIMARY KEY,
    actividad_id INT,
    tipo_factor VARCHAR(120) NOT NULL,
    valor FLOAT NOT NULL,
    FOREIGN KEY (actividad_id) REFERENCES Actividad(actividad_id)
);
